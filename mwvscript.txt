function loadAssetText(path) {
    var is  = ctx.getAssets().open(path);
    var sc  = new java.util.Scanner(is).useDelimiter("\\A");
    var txt = sc.hasNext() ? String(sc.next()) : "";
    sc.close();
    is.close();
    return txt;
}

function readFileText(filePath) {
    var file = new java.io.File(filePath);
    if (!file.exists()) return "";
    var fis  = new java.io.FileInputStream(file);
    var sc2  = new java.util.Scanner(fis, "UTF-8").useDelimiter("\\A");
    var txt  = sc2.hasNext() ? String(sc2.next()) : "";
    sc2.close();
    fis.close();
    return txt;
}

function jsoupToJson(Jsoup, assetPath) {
    try {
        var html = loadAssetText(assetPath);
        var doc  = Jsoup.parse(html);
        return JSON.stringify({ title: String(doc.title()), body: String(doc.body().html()) });
    } catch(e) {
        return JSON.stringify({ title: assetPath, body: "" });
    }
}

function convertToRJS(code, engine) {
    try {
        if (engine === "typescript") return String(ts.transpile(code));
        else if (engine === "livescript") return String(LiveScript.compile(code));
        else if (engine === "babel") {
            var result = Babel.transform(code, {presets: ["es2015"]});
            return String(result.code);
        } else if (engine === "html") {
            var Jsoup = Packages.org.jsoup.Jsoup;
            var doc   = Jsoup.parse(code);
            return String(doc.select("script").html());
        }
        return code;
    } catch(e) {
        return "// 変換エラー: " + e.message + "\n" + code;
    }
}

function sanitizeAndroidRefs(code) {
    return code.replace(/([^.\w])android\.([A-Za-z])/g, function(match, prefix, nextChar) {
        return prefix + "Packages.android." + nextChar;
    });
}

function runTerminal() {
    var code = String(terminalInput.getText().toString().trim());
    if (code.length === 0) return;
    var risk = scanScript(code);
    if (risk.blocked) { terminalAppend("[BLOCKED] " + risk.reason); return; }
    if (risk.needsBiometric) {
        var confirmed = new pkg.utils.Confirm(ctx, "実行確認",
            "以下の操作が含まれます:\n\n" + risk.summary + "\n\n実行しますか?", "実行", "キャンセル");
        confirmed.waitUntilDismiss();
        if (!confirmed.isConfirmed()) { terminalAppend("[CANCELLED]"); return; }
    }
    terminalAppend("[" + terminalEngine.toUpperCase() + "] > " + code.substring(0, 60));
    var intent = new Intent();
    if (terminalEngine === "bsh") {
        intent.setClassName("com.completeapps.jascriptapp", "com.completeapps.jascriptapp.BeanShellActivity");
    } else {
        intent.setClassName("com.completeapps.jascriptapp", "com.completeapps.jascriptapp.ScriptActivity");
    }
    intent.putExtra("script", code);
    intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
    ctx.startActivity(intent);
    terminalAppend("[sent to JAS engine]");
}

function terminalAppend(text) {
    if (terminalOutput == null) return;
    terminalOutput.setText(String(terminalOutput.getText().toString()) + text + "\n");
}

function loadApps() {
    var pm = ctx.getPackageManager();
    var intent = new Intent(Intent.ACTION_MAIN, null);
    intent.addCategory(Intent.CATEGORY_LAUNCHER);
    var apps = pm.queryIntentActivities(intent, 1);
    launcherLabel = []; launcherPkg = []; launcherPkgName = [];
    for (var i = 0; i < apps.size(); i++) {
        launcherLabel.push(apps.get(i).activityInfo.loadLabel(pm).toString());
        launcherPkg.push(apps.get(i).activityInfo.loadIcon(pm));
        launcherPkgName.push(apps.get(i).activityInfo.packageName);
    }
}

function launchApp(position) {
    var intent = ctx.getPackageManager().getLaunchIntentForPackage(launcherPkgName[position]);
    if (intent != null) ctx.startActivity(intent);
    else popup("起動できません");
}

function openInEdit(relativePath) {
    var file = new java.io.File(SAMPLES_BASE + relativePath);
    if (!file.exists()) { popup("テンプレートが見つかりません:\n" + relativePath); return; }
    var intent = new Intent(Intent.ACTION_EDIT);
    intent.setDataAndType(Uri.fromFile(file), "text/plain");
    intent.setPackage("com.completeapps.jascriptapp");
    try { ctx.startActivity(intent); } catch(e) { popup("JASが見つかりません"); }
}

function runTemplate(path) {
    var fullPath = path.indexOf("/") === 0 ? path : SAMPLES_BASE + path;
    var file = new java.io.File(fullPath);
    if (!file.exists()) { popup("ファイルが見つかりません:\n" + fullPath); return; }

    var code = readFileText(fullPath);
    var engine = "rjs";
    if (fullPath.indexOf(".ts") > -1)        engine = "typescript";
    else if (fullPath.indexOf(".ls") > -1)   engine = "livescript";
    else if (fullPath.indexOf(".py") > -1)   engine = "brython";
    else if (fullPath.indexOf(".html") > -1) engine = "html";
    else if (fullPath.indexOf(".js") > -1)   engine = "babel";
    else if (fullPath.indexOf(".bsh") > -1)  engine = "bsh";
    else if (fullPath.indexOf(".java") > -1) engine = "bsh";
    else if (fullPath.indexOf(".kt") > -1)   engine = "kotlite";

    if (engine !== "rjs" && engine !== "bsh") code = convertToRJS(code, engine);
    if (engine === "rjs") code = sanitizeAndroidRefs(code);

    var risk = scanScript(code);
    if (risk.blocked) { popup("実行禁止パターンを検出:\n" + risk.reason); return; }
    if (risk.needsBiometric) {
        var confirmed = new pkg.utils.Confirm(ctx, "実行確認",
            "以下の操作が含まれます:\n\n" + risk.summary + "\n\n実行しますか?", "実行", "キャンセル");
        confirmed.waitUntilDismiss();
        if (!confirmed.isConfirmed()) return;
    }

    if (engine === "html") {
        updateTabColors(TAB_WEB); showTab(TAB_WEB);
        runOnUIThread(function() {
            if (webView != null) webView.loadDataWithBaseURL("file:///android_asset/", code, "text/html", "UTF-8", null);
        });
        return;
    }

    var intent = new Intent();
    if (engine === "bsh") {
        intent.setClassName("com.completeapps.jascriptapp", "com.completeapps.jascriptapp.BeanShellActivity");
        intent.putExtra("script", code);
        ctx.startActivity(intent);
    } else {
        intent.setClassName("com.completeapps.jascriptapp", "com.completeapps.jascriptapp.ScriptActivity");
        intent.putExtra("script", code);
        intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
        ctx.startActivity(intent);
    }
}

// ASTスキャン
// ============================================================
function scanScript(script) {
    var result = { blocked: false, needsBiometric: false, reason: "", summary: "" };
    var risks = [];
    for (var i = 0; i < BLOCKED_PATTERNS.length; i++) {
        if (BLOCKED_PATTERNS[i][0].test(script)) {
            result.blocked = true; result.reason = BLOCKED_PATTERNS[i][1]; return result;
        }
    }
    for (var j = 0; j < RISK_PATTERNS.length; j++) {
        if (RISK_PATTERNS[j][0].test(script)) risks.push(RISK_PATTERNS[j][1]);
    }
    if (risks.length > 0) { result.needsBiometric = true; result.summary = risks.join("\n"); }
    return result;
}

function setupRecorderNotification() {
    var nm = ctx.getSystemService(AppContext.NOTIFICATION_SERVICE);
    if (Packages.android.os.Build.VERSION.SDK_INT >= 26) {
        var ch = new NotificationChannel(RECORDER_CHANNEL, "マクロレコーダー", NotificationMgr.IMPORTANCE_LOW);
        nm.createNotificationChannel(ch);
    }
    var pi = PendingIntent.getActivity(ctx, 0, new Intent(ctx, pkg.ScriptActivity),
        PendingIntent.FLAG_IMMUTABLE | PendingIntent.FLAG_UPDATE_CURRENT);
    var b = new AppNotification.Builder(ctx)
        .setSmallIcon(pkg.R.drawable.edit)
        .setContentTitle("MWV マクロレコーダー")
        .setContentText("タップして通知パネルを閉じると録画開始")
        .setPriority(-1).setOngoing(true).setContentIntent(pi);
    if (Packages.android.os.Build.VERSION.SDK_INT >= 26) b.setChannelId(RECORDER_CHANNEL);
    nm.notify(RECORDER_NOTIF_ID, b.build());
}
